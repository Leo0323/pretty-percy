<script>
/* ========== 基础状态 ========== */
let feedMode = 'latest';     // latest | hot
let feedPage = 1;
const pageSize = 6;
let totalPages = 1;
let currentKeyword = '';  // 搜索关键词
let currentType = '';     // 类型筛选

const latestBtn = document.getElementById('latestBtn');
const hotBtn = document.getElementById('hotBtn');
const fileGrid = document.getElementById('fileGrid');
const feedPageText = document.getElementById('feedPageText');
const indicator = document.getElementById('indicator');

/* ========== 登录态 ========== */
/* ========= GitHub 回调 token 处理 ========= */
(function () {
    const params = new URLSearchParams(window.location.search);
    const token = params.get("token");
    const username = params.get("username");
	const userId = params.get("userId");
    if (token) {
        const user = {
			userId:Number(userId),
            token: token,
            userName: username
        };

        localStorage.setItem("loggedInUser", JSON.stringify(user));

        // 清除 URL 参数（防止刷新重复登录）
        window.history.replaceState({}, document.title, window.location.pathname);

        console.log("GitHub 登录成功", user);
    }
})();

/* ========= 登录态校验 ========= */
const loggedInUser = JSON.parse(localStorage.getItem("loggedInUser"));

if (!loggedInUser || !loggedInUser.token) {
    window.location.href = "login.html";
}

/* ========== 加载动态 ========== */
function loadFeed() {
    let url = '';
    const params = {
        currentPage: feedPage,
        pageSize: pageSize
    };

    if (currentKeyword) params.name = currentKeyword;
    if (currentType) params.type = currentType;

    if (feedMode === 'latest') {
        url = 'http://localhost:820/robert/getFileByTime';
    } else {
        url = 'http://localhost:820/robert/getFileByLike';
    }

    axios.get(url, {
        params,
        headers: { Authorization: loggedInUser.token }
    }).then(res => {
        const page = res.data;

        const files = page.records || [];
        totalPages = page.pages || 1;

        renderFiles(files);
        feedPageText.innerText = `第 ${feedPage} / ${totalPages} 页`;
    }).catch(err => {
        console.error("loadFeed error", err);
  });
}
/* ========== 渲染动态文件列表 ========== */
function renderFiles(files) {
    fileGrid.innerHTML = "";
    if (!files || files.length === 0) {
        fileGrid.innerHTML = `<div class="empty">暂无动态</div>`;
        return;
    }

files.forEach(file => {
    const card = document.createElement("div");
    card.className = "feed-card";

    const mediaContainer = document.createElement("div");
    mediaContainer.className = "img-box";

    // ===== 媒体渲染 =====
    if (file.fileType === "image" || file.fileType === "video") {

        if (file.ossUrl) {
            // OSS 直接渲染
            if (file.fileType === "image") {
                const img = new Image();
                img.src = file.ossUrl;
                img.alt = file.fileName;
                mediaContainer.appendChild(img);
            } else {
                const video = document.createElement("video");
                video.src = file.ossUrl;
                video.controls = true;
                mediaContainer.appendChild(video);
            }
            appendCaption(mediaContainer, file);

        } else {
            // 本地磁盘通过 axios 异步获取 blob
            axios.get(
                `http://localhost:820/robert/getFileById/${file.id}`,
                { responseType: "blob", headers: { Authorization: loggedInUser.token } }
            ).then(res => {
                const url = URL.createObjectURL(res.data);

                if (file.fileType === "image") {
                    const img = new Image();
                    img.src = url;
                    img.alt = file.fileName;
                    mediaContainer.appendChild(img);
                } else {
                    const video = document.createElement("video");
                    video.src = url;
                    video.controls = true;
                    mediaContainer.appendChild(video);
                }

                appendCaption(mediaContainer, file);
            }).catch(err => {
                console.error("加载媒体失败", err);
                appendCaption(mediaContainer, file);
            });
        }

    } else {
        // 非媒体文件
        const colors = ["#8ecae6", "#a8dadc", "#ffd166", "#cdb4db", "#b7e4c7"];
        mediaContainer.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
        mediaContainer.style.display = "flex";
        mediaContainer.style.alignItems = "center";
        mediaContainer.style.justifyContent = "center";
        appendCaption(mediaContainer, file);
    }

    card.appendChild(mediaContainer);
    fileGrid.appendChild(card);

        // ===== 点赞区 =====
        const likeDiv = document.createElement("div");
        likeDiv.className = "feed-meta";
        const likeIcon = document.createElement("img");
        likeIcon.src = "./img/灰色.png";
        likeIcon.alt = "点赞";
        likeDiv.appendChild(likeIcon);

        let liked = false;
		const user = JSON.parse(localStorage.getItem("loggedInUser"));
		console.log("login success",user);
        axios.get(`http://localhost:820/robert/like/hasLiked/${file.id}/${user.userId}`, {
            headers: { Authorization: user.token }
        }).then(res => {
            liked = res.data === true;
            if (liked) {
				likeIcon.src = "./img/huangse1.png";
			likeIcon.classList.add("liked");
			}
			
        });

       likeIcon.addEventListener("click", e => {
           e.stopPropagation();
       
		    const user = JSON.parse(localStorage.getItem("loggedInUser"));
            const url = liked
               ? `http://localhost:820/robert/like/cancel/check/${file.id}/${user.userId}`
               : `http://localhost:820/robert/like/check/${file.id}/${user.userId}`;
       
           axios.get(url, { headers: { Authorization: user.token } })
               .then(() => {
                   liked = !liked;
       
                   if (liked) {
                       likeIcon.src = "./img/huangse1.png";
                       likeIcon.classList.add("liked");
                   } else {
                       likeIcon.src = "./img/灰色.png";
                       likeIcon.classList.remove("liked");
                   }
               })
               .catch(err => console.error("点赞操作失败", err));
       });

        card.appendChild(mediaContainer);
        card.appendChild(likeDiv);

        // ===== 编辑按钮 =====
        if (file.isOwner) {
            const editBtn = document.createElement("button");
            editBtn.innerText = "修改";
            editBtn.onclick = e => {
                e.stopPropagation();
                localStorage.setItem("editFile", JSON.stringify(file));
                window.location.href = "editFile.html";
            };
            card.appendChild(editBtn);
        }

        // ===== 点击跳详情页 =====
        card.addEventListener("click", () => {
            localStorage.setItem("currentFile", JSON.stringify(file));
            //window.location.href = "fileDetail.html";
                 window.open(`fileDetail.html?id=${file.id}`, "_blank");
                // 方式2：替换当前页面（不会在浏览器历史中留下记录，按需选择）
                // window.location.replace("fileDetail.html");

        });

        fileGrid.appendChild(card);
    });
}


/* ========== 附加标题/描述 ====== */
function appendCaption(container, file) {
    const transparentBox = document.createElement("div");
    transparentBox.className = "transparent-box";
    const caption = document.createElement("div");
    caption.className = "caption";
    const titleP = document.createElement("p");
    titleP.textContent = file.fileName;
    const subP = document.createElement("p");
    subP.textContent = file.describtion;
    caption.appendChild(titleP);
    caption.appendChild(subP);
    transparentBox.appendChild(caption);
    container.appendChild(transparentBox);
}

/* ========== 分页 ========== */
function changeFeedPage(step) {
    const next = feedPage + step;
    if (next < 1 || next > totalPages) return;
    feedPage = next;
    loadFeed();
}

/* ========== 最新/最热切换 ========== */
latestBtn.onclick = () => {
    if (feedMode === 'latest') return;
    feedMode = 'latest';
    feedPage = 1;
    latestBtn.classList.add("active");
    hotBtn.classList.remove("active");
    updateIndicator();
    loadFeed();
};

hotBtn.onclick = () => {
    if (feedMode === 'hot') return;
    feedMode = 'hot';
    feedPage = 1;
    hotBtn.classList.add("active");
    latestBtn.classList.remove("active");
    updateIndicator();
    loadFeed();
};

/* ========== 指示条 ====== */
function updateIndicator() {
    const active = feedMode === 'latest' ? latestBtn : hotBtn;
    indicator.style.width = active.offsetWidth + "px";
    indicator.style.transform = `translateX(${active.offsetLeft}px)`;
}

/* ========== 搜索 ====== */
function searchFiles() {
    currentKeyword = document.getElementById("searchInput").value.trim();
    currentType = document.getElementById("typeSelect").value;
    feedPage = 1;
    loadFeed();
}
searchInput.addEventListener("keydown", function (e) {
    if (e.key === "Enter" || e.keyCode === 13) {
        searchFiles();
    }
});

/* ========== 初始化 ====== */
updateIndicator();
loadFeed();
</script>
